SUBMODULE: factorial
IMPORT: number(integer)
EXPORT: factorial(integer)
ASSERTION: 
ALGORITHM:
	factorial = 1
	FOR number DOWN TO 1
		factorial = number * factorial
	ENDFOR

IMPORT: factorial(integer)

ALGORITHM:
	OUTPUT "factorial is " + factorial
